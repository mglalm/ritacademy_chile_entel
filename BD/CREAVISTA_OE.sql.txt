
  CREATE OR REPLACE FORCE EDITIONABLE VIEW "C##TRAIN3"."OE_ORDENVIEW" ("ORDER_ID", "PO_ID", "NUMERO_INTENTO", "PARAM_ID", "VALOR", "RUT", "NOMBRE", "DIRECCION", "COMUNA", "CIUDAD", "CWDOCID", "ESTADO", "MOTIVO_ESTADO", "FECHA_CREACION", "CWDOCSTAMP", "CWORDERCREATIONDATE", "CWPARENTID", "CWSITEID", "LASTUPDATEDDATE", "UPDATEDBY") AS 
  SELECT 
    OM_ORDEN.ORDER_ID,"PO_ID","NUMERO_INTENTO","PARAM_ID","VALOR",OE_CLIENTE.RUT,"NOMBRE","DIRECCION","COMUNA","CIUDAD" ,"CWDOCID","ESTADO","MOTIVO_ESTADO","FECHA_CREACION","CWDOCSTAMP","CWORDERCREATIONDATE","CWPARENTID","CWSITEID","LASTUPDATEDDATE","UPDATEDBY"
 
FROM 
    OE_CLIENTE, OM_ORDEN, OM_PARAM_BSCS
    
WHERE OE_CLIENTE.RUT = OM_ORDEN.RUT and OM_ORDEN.ORDER_ID = OM_PARAM_BSCS.ORDER_ID;


SELECT 
    b.ORDER_ID,c.PO_ID, a.RUT, a.NOMBRE, a.DIRECCION, a.COMUNA, a.CIUDAD , b.CWDOCID, b.ESTADO, b.MOTIVO_ESTADO, d.valor, b.FECHA_CREACION
 
FROM 
    OE_CLIENTE a, OM_ORDEN b, OM_ORDER_PO c, OM_RECURSOS_PO d
    
WHERE a.RUT = b.RUT 
and b.cwDocId = c.cwDocId
and c.PO_Id = d.PO_Id
and c.Order_Id = d.ORDER_ID
and d.RESOURCE_ID = 'MSISDN';
